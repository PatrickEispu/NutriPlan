public int criarAlimento(int idCategoriaAlimento, double kcal, double carboidrato, double proteina, double gordura, double quantidade, String nome) {
    String sql = "SELECT criar_alimento(?, ?, ?, ?, ?, ?, ?)";
    return jdbcTemplate.queryForObject(sql, Integer.class, idCategoriaAlimento, kcal, carboidrato, proteina, gordura, quantidade, nome);
}

public Alimento buscarAlimentoPorId(int idAlimento) {
    String sql = "SELECT * FROM buscar_alimento_por_id(?)";
    return jdbcTemplate.queryForObject(sql, new BeanPropertyRowMapper<>(Alimento.class), idAlimento);
}

public List<Alimento> listarTodosAlimentos() {
    String sql = "SELECT * FROM listar_todos_alimentos()";
    return jdbcTemplate.query(sql, new BeanPropertyRowMapper<>(Alimento.class));
}


public void atualizarAlimento(int idAlimento, int idCategoriaAlimento, double kcal, double carboidrato, double proteina, double gordura, double quantidade, String nome) {
    String sql = "SELECT atualizar_alimento(?, ?, ?, ?, ?, ?, ?, ?)";
    jdbcTemplate.update(sql, idAlimento, idCategoriaAlimento, kcal, carboidrato, proteina, gordura, quantidade, nome);
}

public void deletarAlimento(int idAlimento) {
    String sql = "SELECT deletar_alimento(?)";
    jdbcTemplate.update(sql, idAlimento);
}
