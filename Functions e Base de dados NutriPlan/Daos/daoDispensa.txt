public int criarDispensa(int idConta) {
    String sql = "SELECT criar_dispensa(?)";
    return jdbcTemplate.queryForObject(sql, Integer.class, idConta);
}

public Dispensa buscarDispensaPorId(int idDispensa) {
    String sql = "SELECT * FROM buscar_dispensa_por_id(?)";
    return jdbcTemplate.queryForObject(sql, new BeanPropertyRowMapper<>(Dispensa.class), idDispensa);
}

public void deletarDispensa(int idDispensa) {
    String sql = "SELECT deletar_dispensa(?)";
    jdbcTemplate.update(sql, idDispensa);
}
