public int criarConta(String nome, String email, String senha, int idTipoConta) {
    String sql = "SELECT criar_conta(?, ?, ?, ?)";
    return jdbcTemplate.queryForObject(sql, Integer.class, nome, email, senha, idTipoConta);
}

public Conta buscarContaPorId(int idConta) {
    String sql = "SELECT * FROM buscar_conta_por_id(?)";
    return jdbcTemplate.queryForObject(sql, new BeanPropertyRowMapper<>(Conta.class), idConta);
}

public void atualizarConta(int idConta, String nome, String email, String senha, int idTipoConta) {
    String sql = "SELECT atualizar_conta(?, ?, ?, ?, ?)";
    jdbcTemplate.update(sql, idConta, nome, email, senha, idTipoConta);
}

public void deletarConta(int idConta) {
    String sql = "SELECT deletar_conta(?)";
    jdbcTemplate.update(sql, idConta);
}
