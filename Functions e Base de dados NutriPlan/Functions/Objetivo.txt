CREATE OR REPLACE FUNCTION buscar_objetivo_por_descricao(
    p_ds_objetivo VARCHAR
)
RETURNS TABLE(
    nr_id_objetivo INT,
    ds_objetivo VARCHAR
) AS $$
BEGIN
    RETURN QUERY
    SELECT objetivo.nr_id_objetivo, objetivo.ds_objetivo
    FROM objetivo
    WHERE objetivo.ds_objetivo = p_ds_objetivo;
END;
$$ LANGUAGE plpgsql;
