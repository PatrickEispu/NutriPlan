CREATE OR REPLACE FUNCTION buscar_categoria_por_nome(
    p_nm_categoria VARCHAR
)
RETURNS TABLE(
    nr_id_categoria INT,
    nm_categoria VARCHAR
) AS $$
BEGIN
    RETURN QUERY
    SELECT c.nr_id_categoria, c.nm_categoria
    FROM categoria c
    WHERE c.nm_categoria = p_nm_categoria;
END;
$$ LANGUAGE plpgsql;
